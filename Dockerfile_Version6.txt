# Dockerfile for investmenttracker1.2
FROM python:3.11-slim

ENV PYTHONUNBUFFERED=1
WORKDIR /app

# Install build deps needed for some packages (aiosqlite may need a compiler)
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    && rm -rf /var/lib/apt/lists/*

# Copy dependency list and install
COPY requirements.txt /app/requirements.txt
RUN pip install --no-cache-dir -r /app/requirements.txt

# Copy app
COPY . /app

# Ensure a default persistent DB directory exists (Render persistent disk is usually mounted at /data)
RUN mkdir -p /data

# Default DB path if none provided by env
ENV TRACKER_DB=/data/tracker.db
ENV PORT=8000

EXPOSE 8000

# Use proxy-headers for getting real client IPs behind Render/proxies
CMD ["uvicorn", "tracker_app:app", "--host", "0.0.0.0", "--port", "8000", "--proxy-headers"]
